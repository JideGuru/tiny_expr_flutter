# The Flutter tooling requires that developers have CMake 3.10 or later
# installed. You should not increase this version, as doing so will cause
# the plugin to fail to compile for some customers of the plugin.
cmake_minimum_required(VERSION 3.10)

project(tiny_expr_flutter_library VERSION 0.0.1 LANGUAGES C)

add_library(tiny_expr_flutter SHARED
  "tiny_expr_flutter.c"
)

set_target_properties(tiny_expr_flutter PROPERTIES
  PUBLIC_HEADER tiny_expr_flutter.h
  OUTPUT_NAME "tiny_expr_flutter"
)

target_compile_definitions(tiny_expr_flutter PUBLIC DART_SHARED_LIB)
